#include <stdio.h>

int main() {
    int food = 5, score = 0;
    char move;

    printf("Collect %d food items by pressing 'f'\n", food);
    while(food > 0) {
        printf("Press 'f' to collect: ");
        move = getchar();
        getchar();  // Clear newline

        if (move == 'f') {
            printf("Food collected!\n");
            score++;
            food--;
        } else {
            printf("Wrong key! Try again.\n");
        }
    }
    printf("All food collected! Final Score: %d\n", score);
    return 0;
}
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    char move;
    int caught = 0;
    srand(time(0));

    for(int i=0; i<5; i++) {
        printf("A star is falling! Type 'c' to catch: ");
        move = getchar();
        getchar();  // Clear newline
        if (move == 'c') {
            printf("Caught it!\n");
            caught++;
        } else {
            printf("Missed!\n");
        }
    }
    printf("You caught %d stars!\n", caught);
    return 0;
}
